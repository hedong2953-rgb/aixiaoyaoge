// ===== 搜索引擎切换 ===== const engines = { site: '', bing: 'https://cn.bing.com/search?q=%s', baidu: 'https://www.baidu.com/s?wd=%s', google: 'https://www.google.com/search?q=%s' }; let currentEngine = 'site'; document.querySelectorAll('.search-type li').forEach(el => { el.addEventListener('click', () => { document.querySelectorAll('.search-type li').forEach(e => e.classList.remove('s-current')); el.classList.add('s-current'); currentEngine = el.dataset.engine; }); }); // 搜索提交 document.getElementById('search-text').addEventListener('keydown', e => { if (e.key === 'Enter') { const q = e.target.value.trim(); if (!q) return; if (currentEngine === 'site') { doSearch(); return; } window.open(engines[currentEngine].replace('%s', encodeURIComponent(q)), '_blank'); } });
